    <script>

        document.addEventListener('DOMContentLoaded', function() {

            var calendarEl = document.getElementById('calendar');

            var calendar = new FullCalendar.Calendar(calendarEl, {

                initialView: 'dayGridMonth',

                selectable: true,

                editable: true,

                events: []

            });

            calendar.render();

        });

        

        function countdown() {

            const targetDate = new Date("2025-06-01").getTime();

            const now = new Date().getTime();

            const difference = targetDate - now;

            

            if (difference < 0) {

                document.getElementById("timer").innerText = "We're Together Now! 💖";

                return;

            }

            

            const days = Math.floor(difference / (1000 * 60 * 60 * 24));

            document.getElementById("timer").innerText = days + " days left!";

        }

        

        function saveNotes() {

            localStorage.setItem("notes", document.getElementById("notes").value);

        }

        

        function loadNotes() {

            document.getElementById("notes").value = localStorage.getItem("notes") || "";

        }

        

        function generateQuestion() {

            const questions = [

                "What is one thing you love about us?",

                "What was your favorite moment with me?",

                "What is one dream we share?",

                "How can we make our bond stronger?",

                "What is something new you'd like to try together?"

            ];

            const today = new Date().getDate();

            document.getElementById("daily-question").innerText = questions[today % questions.length];

        }

        

        countdown();

        setInterval(countdown, 1000 * 60 * 60 * 24);

        loadNotes();

        generateQuestion();

    </script>
